### movies-app

CocoaPods

# Podfile
use_frameworks!

target 'YOUR_TARGET_NAME' do
    pod 'SwiftyJSON'
    pod 'RealmSwift'
end

Replace YOUR_TARGET_NAME and then, in the Podfile directory, type:

$ pod install
